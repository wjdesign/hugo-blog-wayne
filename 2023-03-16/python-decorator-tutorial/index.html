def retry(max=1):
  class Wrapper:
    def __init__(self, func):
      self.func = func

    def __call__(self):
      retried = 0
      while retried < max:
        try:
          self.func()
        except Exception:
          retried += 1
          print('Failed. Going to try again (', retried, ')')
        else:
          break
  return Wrapper
