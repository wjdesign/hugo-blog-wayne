async function isSupportWebp() {
  return new Promise(resolve => {
    let result = false
    const elem = document.createElement("canvas")
    if (elem.getContext && elem.getContext("2d")) {
      result = elem.toDataURL("image/webp").indexOf("data:image/webp") === 0
    }
    resolve(result)
  })
}
